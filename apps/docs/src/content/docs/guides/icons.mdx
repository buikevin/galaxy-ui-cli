## Basic Usage

### 1. Import the Module

Add `LucideAngularModule` to your component imports:

```typescript
import { Component } from '@angular/core';
import { LucideAngularModule } from 'lucide-angular';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [LucideAngularModule],
  template: `
    <lucide-angular [img]="'home'" class="w-6 h-6"></lucide-angular>
  `
})
export class ExampleComponent {}
```

### 2. Use Icons in Templates

```html
<!-- Basic icon -->
<lucide-angular [img]="'home'" class="w-5 h-5"></lucide-angular>

<!-- With custom size -->
<lucide-angular [img]="'user'" class="w-8 h-8"></lucide-angular>

<!-- With color -->
<lucide-angular [img]="'check'" class="w-4 h-4 text-green-600"></lucide-angular>

<!-- Dynamic icon -->
<lucide-angular [img]="iconName()" class="w-6 h-6"></lucide-angular>
```

### 3. Dynamic Icons

```typescript
import { Component, signal } from '@angular/core';
import { LucideAngularModule, icons } from 'lucide-angular';

@Component({
  imports: [LucideAngularModule],
  template: `
    <lucide-angular [img]="currentIcon()" class="w-6 h-6"></lucide-angular>
    <button (click)="toggleIcon()">Toggle</button>
  `
})
export class IconExample {
  currentIcon = signal<keyof typeof icons>('sun');

  toggleIcon() {
    this.currentIcon.update(icon => icon === 'sun' ? 'moon' : 'sun');
  }
}
```

## Icon Sizes

Use Tailwind classes for consistent sizing:

```html
<!-- Extra small -->
<lucide-angular [img]="'star'" class="w-3 h-3"></lucide-angular>

<!-- Small -->
<lucide-angular [img]="'star'" class="w-4 h-4"></lucide-angular>

<!-- Medium (default) -->
<lucide-angular [img]="'star'" class="w-5 h-5"></lucide-angular>

<!-- Large -->
<lucide-angular [img]="'star'" class="w-6 h-6"></lucide-angular>

<!-- Extra large -->
<lucide-angular [img]="'star'" class="w-8 h-8"></lucide-angular>
```

| Size | Class | Pixels |
|------|-------|--------|
| XS | `w-3 h-3` | 12px |
| SM | `w-4 h-4` | 16px |
| MD | `w-5 h-5` | 20px |
| LG | `w-6 h-6` | 24px |
| XL | `w-8 h-8` | 32px |
| 2XL | `w-10 h-10` | 40px |

## Colors

Apply colors using Tailwind text color classes:

```html
<!-- Gray -->
<lucide-angular [img]="'circle'" class="w-5 h-5 text-slate-500"></lucide-angular>

<!-- Primary -->
<lucide-angular [img]="'circle'" class="w-5 h-5 text-violet-600"></lucide-angular>

<!-- Warning -->

<!-- Danger -->

</lucide-angular>
```

## Using Icons with Galaxy UI Components

### With Buttons

```html
<button gButton [buttonStatus]="'primary'">
  <lucide-angular [img]="'plus'" class="w-4 h-4 mr-2"></lucide-angular>
  Add Item
</button>

<button gButton [buttonAppearance]="'outline'">
  <lucide-angular [img]="'trash'" class="w-4 h-4 mr-2"></lucide-angular>
  Delete
</button>

<!-- Icon only button -->
<button gButton [buttonShape]="'round'" [buttonSize]="'small'">
  <lucide-angular [img]="'search'" class="w-4 h-4"></lucide-angular>
</button>
```

### With Menu

```html
<g-menu>
  <g-menu-item [itemIcon]="'home'" [itemLabel]="'Home'"></g-menu-item>
  <g-menu-item [itemIcon]="'inbox'" [itemLabel]="'Messages'"></g-menu-item>
  <g-menu-item [itemIcon]="'settings'" [itemLabel]="'Settings'"></g-menu-item>
</g-menu>
```

### With Input

```html
<g-input
  [inputLabel]="'Search'"
  [inputPrefixIcon]="'search'"
  [inputPlaceholder]="'Search...'">
</g-input>
```

### With Card

```html
<g-card>
  <g-card-header>
    <div class="flex items-center gap-2">
      <lucide-angular [img]="'users'" class="w-5 h-5 text-violet-600"></lucide-angular>
      <h3>Team Members</h3>
    </div>
  </g-card-header>
  <g-card-body>
    Content here
  </g-card-body>
</g-card>
```

## Advanced Usage

### Animated Icons

```html
<!-- Spinning loader -->
<lucide-angular
  [img]="'loader-2'"
  class="w-5 h-5 animate-spin text-violet-600">
</lucide-angular>

<!-- Pulse animation -->
<lucide-angular
  [img]="'bell'"
  class="w-5 h-5 animate-pulse text-amber-500">
</lucide-angular>

<!-- Custom animation -->
<lucide-angular
  [img]="'heart'"
  class="w-5 h-5 transition-all duration-200 hover:scale-110 hover:text-red-500">
</lucide-angular>
```

### Icon Buttons

```typescript
@Component({
  template: `
    <div class="flex gap-2">
      @for (action of actions; track action.name) {
        <button
          gButton
          [buttonShape]="'round'"
          [buttonSize]="'small'"
          [buttonAppearance]="'ghost'"
          [title]="action.name"
          (click)="handleAction(action.name)">
          <lucide-angular [img]="action.icon" class="w-4 h-4"></lucide-angular>
        </button>
      }
    </div>
  `
})
export class IconButtonsExample {
  actions = [
    { name: 'Edit', icon: 'pencil' as const },
    { name: 'Copy', icon: 'copy' as const },
    { name: 'Delete', icon: 'trash-2' as const },
  ];

  handleAction(name: string) {
    console.log('Action:', name);
  }
}
```

### Conditional Icons

```typescript
@Component({
  template: `
    <button gButton (click)="toggleFavorite()">
      <lucide-angular
        [img]="isFavorite() ? 'heart' : 'heart'"
        [class]="isFavorite() ? 'w-5 h-5 text-red-500 fill-red-500' : 'w-5 h-5'">
      </lucide-angular>
      {{ isFavorite() ? 'Favorited' : 'Add to favorites' }}
    </button>
  `
})
export class FavoriteButtonExample {
  isFavorite = signal(false);

  toggleFavorite() {
    this.isFavorite.update(v => !v);
  }
}
```

## Best Practices

1. **Consistent Sizing**: Use the same icon size within a context
   ```html
   <!-- Good -->
   <lucide-angular [img]="'home'" class="w-5 h-5"></lucide-angular>
   <lucide-angular [img]="'settings'" class="w-5 h-5"></lucide-angular>

   <!-- Bad -->
   <lucide-angular [img]="'home'" class="w-5 h-5"></lucide-angular>
   <lucide-angular [img]="'settings'" class="w-8 h-8"></lucide-angular>
   ```

2. **Semantic Icons**: Use icons that match the action
   ```html
   <!-- Good -->
   <button>
     <lucide-angular [img]="'trash-2'"></lucide-angular>
     Delete
   </button>

   <!-- Bad -->
   <button>
     <lucide-angular [img]="'smile'"></lucide-angular>
     Delete
   </button>
   ```

3. **Accessibility**: Always provide text labels or `aria-label`
   ```html
   <!-- With text -->
   <button gButton>
     <lucide-angular [img]="'save'"></lucide-angular>
     Save
   </button>

   <!-- Icon only - use aria-label -->
   <button gButton aria-label="Save">
     <lucide-angular [img]="'save'"></lucide-angular>
   </button>
   ```

4. **Color Contrast**: Ensure sufficient contrast in dark mode
   ```html
   <lucide-angular
     [img]="'sun'"
     class="w-5 h-5 text-slate-700 dark:text-slate-300">
   </lucide-angular>
   ```

## TypeScript Types

For type safety with dynamic icons:

```typescript
import { icons } from 'lucide-angular';

// Type-safe icon name
type IconName = keyof typeof icons;

// In component
iconName = signal<IconName>('home');

// Type-safe icon list
const myIcons: IconName[] = ['home', 'settings', 'user'];
```

## Troubleshooting

### Icons not showing

1. **Check module import**:
   ```typescript
   imports: [LucideAngularModule]  // ✅ Correct
   imports: []                      // ❌ Missing
   ```

2. **Check icon name**:
   ```html
   [img]="'home'"      ✅ Correct
   [img]="'Home'"      ❌ Wrong case
   [img]="'house'"     ❌ Wrong name (use 'home')
   ```

3. **Check Tailwind classes**:
   ```html
   class="w-5 h-5"    ✅ Both width and height
   class="w-5"        ❌ Missing height
   ```

### Icons too small/large

Adjust using Tailwind size classes:
```html
<!-- Too small? Increase size -->
class="w-6 h-6"  <!-- Instead of w-4 h-4 -->

<!-- Too large? Decrease size -->
class="w-4 h-4"  <!-- Instead of w-8 h-8 -->
```
