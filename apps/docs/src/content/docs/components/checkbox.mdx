The **Checkbox** component provides an accessible and visually appealing checkbox with support for labels, descriptions, and the indeterminate state.

## Features

- Multiple status colors
- Label and description support
- Indeterminate state
- Disabled state
- Custom styling with Tailwind CSS
- Dark mode support

## Installation

```bash
npx galaxy-ui-cli@latest add checkbox
```

## Usage

### Basic Usage

```typescript
import { Component, signal } from '@angular/core';
import { CheckboxComponent } from '@/components/ui/checkbox';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [CheckboxComponent],
  template: `
    <galaxy-checkbox
      [checkboxLabel]="'Accept terms and conditions'"
      [checkboxChecked]="accepted()"
      (checkedChange)="accepted.set($event)">
    </galaxy-checkbox>
  `
})
export class ExampleComponent {
  accepted = signal(false);
}
```

### With Description

### Indeterminate State

```typescript
@Component({
  template: `
    <galaxy-checkbox
      [checkboxLabel]="'Select all'"
      [checkboxChecked]="allSelected()"
      [checkboxIndeterminate]="someSelected()"
      (checkedChange)="toggleAll($event)">
    </galaxy-checkbox>

    @for (item of items(); track item.id) {
      <galaxy-checkbox
        [checkboxLabel]="item.name"
        [checkboxChecked]="item.selected"
        (checkedChange)="item.selected = $event; updateSelectAll()">
      </galaxy-checkbox>
    }
  `
})
export class CheckboxGroupExample {
  items = signal([
    { id: 1, name: 'Item 1', selected: false },
    { id: 2, name: 'Item 2', selected: false },
    { id: 3, name: 'Item 3', selected: false },
  ]);

  allSelected = computed(() =>
    this.items().every(item => item.selected)
  );

  someSelected = computed(() => {
    const selected = this.items().filter(item => item.selected).length;
    return selected > 0 && selected < this.items().length;
  });

  toggleAll(checked: boolean) {
    this.items.update(items =>
      items.map(item => ({ ...item, selected: checked }))
    );
  }

  updateSelectAll() {
    // Triggers computed signals
    this.items.set([...this.items()]);
  }
}
```

### Disabled State

```html
<galaxy-checkbox
  [checkboxLabel]="'Disabled'"
  [checkboxDisabled]="true">
</galaxy-checkbox>

<galaxy-checkbox
  [checkboxLabel]="'Disabled checked'"
  [checkboxChecked]="true"
  [checkboxDisabled]="true">
</galaxy-checkbox>
```

## Props

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `checkboxLabel` | `string` | `''` | Label text |
| `checkboxDescription` | `string` | `''` | Description text below label |
| `checkboxChecked` | `boolean` | `false` | Checked state |
| `checkboxIndeterminate` | `boolean` | `false` | Indeterminate state |
| `checkboxStatus` | `CheckboxStatus` | `'primary'` | Status color |
| `checkboxDisabled` | `boolean` | `false` | Disabled state |

## Events

| Event | Type | Description |
|-------|------|-------------|
| `checkedChange` | `EventEmitter<boolean>` | Emitted when checkbox state changes |

## Accessibility

- Uses semantic checkbox input
- Proper ARIA attributes
- Keyboard accessible
- Focus visible indicator
- Label click support
