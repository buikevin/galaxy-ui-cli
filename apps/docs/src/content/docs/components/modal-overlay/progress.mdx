The **Progress** component displays the current progress of an operation or task. Inspired by Ant Design's Progress component with line, circle, and dashboard variants.

**Progress** là component hiển thị tiến trình hiện tại của một hoạt động hoặc tác vụ. Lấy cảm hứng từ Progress component của Ant Design với các biến thể dạng đường, vòng tròn, và bảng điều khiển.

## Features | Tính Năng

- **3 types** | **3 loại**: Line, circle, dashboard
- **Status colors** | **Màu trạng thái**: Success, error, warning, info
- **Percentage display** | **Hiển thị phần trăm**: Show/hide completion percentage
- **Custom colors** | **Màu tùy chỉnh**: Gradient and solid color support
- **Format function** | **Hàm định dạng**: Custom percentage text
- **Animated** | **Hoạt ảnh**: Smooth transition animations

### Custom Format | Định Dạng Tùy Chỉnh

```typescript
import { Component, signal } from '@angular/core';

@Component({
  template: `
    <!-- Custom text format | Định dạng văn bản tùy chỉnh -->
    <g-progress
      [percent]="percent()"
      [format]="formatPercent" />

    <!-- Time remaining | Thời gian còn lại -->
    <g-progress
      [percent]="uploadProgress()"
      [format]="formatTime" />

    <!-- File size | Kích thước tệp -->
    <g-progress
      [percent]="downloadProgress()"
      [format]="formatSize" />
  `
})
export class CustomFormatExample {
  percent = signal(75);
  uploadProgress = signal(45);
  downloadProgress = signal(60);

  formatPercent = (percent: number) => {
    return `${percent}% Complete`;
  };

  formatTime = (percent: number) => {
    const remaining = Math.round((100 - percent) / 10);
    return `${remaining}m left`;
  };

  formatSize = (percent: number) => {
    const current = Math.round(percent * 1.5);
    return `${current}MB / 150MB`;
  };
}
```

## Props | Thuộc Tính

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `percent` | `number` | `0` | Progress percentage (0-100) <br/> Phần trăm tiến trình (0-100) |
| `type` | `'line' \| 'circle' \| 'dashboard'` | `'line'` | Progress type <br/> Loại tiến trình |
| `status` | `'normal' \| 'success' \| 'error' \| 'warning'` | `'normal'` | Progress status <br/> Trạng thái tiến trình |
| `strokeColor` | `string \| object` | - | Progress bar color <br/> Màu thanh tiến trình |
| `strokeWidth` | `number` | `8` | Width of progress bar <br/> Độ dày thanh tiến trình |
| `showInfo` | `boolean` | `true` | Show percentage text <br/> Hiển thị phần trăm |
| `format` | `(percent: number) => string` | - | Custom format function <br/> Hàm định dạng tùy chỉnh |
| `width` | `number` | `120` | Circle/dashboard diameter <br/> Đường kính vòng tròn/bảng điều khiển |

- Screen reader announcements for progress updates
- Semantic HTML structure
- High contrast colors for visibility

### File Upload Progress | Tiến Trình Tải Lên

```typescript
import { Component, signal } from '@angular/core';

@Component({
  template: `
    <div class="space-y-4">
      <div class="flex items-center gap-4">
        <div class="flex-1">
          <div class="flex justify-between text-sm mb-2">
            <span>document.pdf</span>
            <span>{{ uploadedSize() }}MB / {{ totalSize() }}MB</span>
          </div>
          <g-progress
            [percent]="uploadPercent()"
            [format]="formatUpload"
            status="normal" />
        </div>
        <button
          (click)="cancelUpload()"
          class="text-red-600 hover:text-red-700">
          Cancel
        </button>
      </div>

      @if (uploadPercent() === 100) {
        <div class="flex items-center gap-2 text-green-600">
          <svg class="w-5 h-5" fill="currentColor" viewBox="0 0 20 20">
            <path fill-rule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clip-rule="evenodd"/>
          </svg>
          <span>Upload complete!</span>
        </div>
      }
    </div>
  `
})
export class FileUploadProgress {
  uploadPercent = signal(0);
  uploadedSize = signal(0);
  totalSize = signal(15);

  constructor() {
    this.simulateUpload();
  }

  formatUpload = (percent: number) => {
    return `${percent}% uploaded`;
  };

  simulateUpload() {
    const interval = setInterval(() => {
      const current = this.uploadPercent();
      if (current >= 100) {
        clearInterval(interval);
        return;
      }

      const increment = Math.random() * 10;
      const newPercent = Math.min(100, current + increment);
      this.uploadPercent.set(Math.round(newPercent));
      this.uploadedSize.set(
        Math.round((newPercent / 100) * this.totalSize())
      );
    }, 500);
  }

  cancelUpload() {
    // Cancel upload logic
    this.uploadPercent.set(0);
    this.uploadedSize.set(0);
  }
}
```

### Profile Completion | Hoàn Thành Hồ Sơ

```typescript
import { Component, computed, signal } from '@angular/core';

@Component({
  template: `
    <div class="bg-white dark:bg-gray-800 rounded-lg p-6">
      <div class="flex items-start gap-4 mb-6">
        <g-progress
          type="circle"
          [percent]="completionPercent()"
          [width]="80"
          [status]="completionStatus()" />

        <div class="flex-1">
          <h3 class="text-lg font-semibold mb-1">
            Complete Your Profile | Hoàn Thành Hồ Sơ
          </h3>
          <p class="text-gray-600 text-sm">
            {{ completionPercent() }}% complete • {{ remainingSteps() }} steps remaining
          </p>
        </div>
      </div>

      <div class="space-y-3">
        <label class="flex items-center gap-3">
          <input
            type="checkbox"
            [(ngModel)]="steps().basicInfo"
            class="w-4 h-4" />
          <span>Add basic information | Thêm thông tin cơ bản</span>
        </label>

        <label class="flex items-center gap-3">
          <input
            type="checkbox"
            [(ngModel)]="steps().avatar"
            class="w-4 h-4" />
          <span>Upload profile photo | Tải ảnh hồ sơ</span>
        </label>

        <label class="flex items-center gap-3">
          <input
            type="checkbox"
            [(ngModel)]="steps().bio"
            class="w-4 h-4" />
          <span>Write bio | Viết tiểu sử</span>
        </label>

        <label class="flex items-center gap-3">
          <input
            type="checkbox"
            [(ngModel)]="steps().social"
            class="w-4 h-4" />
          <span>Connect social accounts | Kết nối tài khoản mạng xã hội</span>
        </label>

        <label class="flex items-center gap-3">
          <input
            type="checkbox"
            [(ngModel)]="steps().verification"
            class="w-4 h-4" />
          <span>Verify email | Xác minh email</span>
        </label>
      </div>
    </div>
  `
})
export class ProfileCompletion {
  steps = signal({
    basicInfo: true,
    avatar: true,
    bio: false,
    social: false,
    verification: false
  });

  completionPercent = computed(() => {
    const stepValues = Object.values(this.steps());
    const completed = stepValues.filter(Boolean).length;
    return Math.round((completed / stepValues.length) * 100);
  });

  remainingSteps = computed(() => {
    const stepValues = Object.values(this.steps());
    return stepValues.filter(v => !v).length;
  });

  completionStatus = computed(() => {
    const percent = this.completionPercent();
    if (percent === 100) return 'success';
    if (percent >= 50) return 'normal';
    return 'warning';
  });
}
```

### Loading Indicator | Chỉ Báo Tải

```typescript
@Component({
  template: `
    <div class="space-y-6">
      <!-- Processing stages | Các giai đoạn xử lý -->
      <div>
        <h3 class="font-semibold mb-4">Processing Order | Xử Lý Đơn Hàng</h3>

        <div class="space-y-4">
          <!-- Stage 1: Validation | Giai đoạn 1: Xác thực -->
          <div>
            <div class="flex justify-between text-sm mb-2">
              <span>Validation | Xác thực</span>
              <span>{{ stages().validation }}%</span>
            </div>
            <g-progress
              [percent]="stages().validation"
              [status]="getStageStatus('validation')" />
          </div>

          <!-- Stage 2: Payment | Giai đoạn 2: Thanh toán -->
          <div>
            <div class="flex justify-between text-sm mb-2">
              <span>Payment Processing | Xử lý thanh toán</span>
              <span>{{ stages().payment }}%</span>
            </div>
            <g-progress
              [percent]="stages().payment"
              [status]="getStageStatus('payment')" />
          </div>

          <!-- Stage 3: Fulfillment | Giai đoạn 3: Thực hiện -->
          <div>
            <div class="flex justify-between text-sm mb-2">
              <span>Fulfillment | Thực hiện</span>
              <span>{{ stages().fulfillment }}%</span>
            </div>
            <g-progress
              [percent]="stages().fulfillment"
              [status]="getStageStatus('fulfillment')" />
          </div>
        </div>
      </div>

      <!-- Overall progress | Tiến trình tổng thể -->
      <div class="border-t pt-6">
        <div class="flex justify-between items-center mb-4">
          <h3 class="font-semibold">Overall Progress | Tiến Trình Tổng Thể</h3>
          <span class="text-2xl font-bold">{{ overallProgress() }}%</span>
        </div>
        <g-progress
          [percent]="overallProgress()"
          [strokeWidth]="12"
          [status]="overallProgress() === 100 ? 'success' : 'normal'" />
      </div>
    </div>
  `
})
export class MultiStageProgress {
  stages = signal({
    validation: 100,
    payment: 60,
    fulfillment: 0
  });

  overallProgress = computed(() => {
    const values = Object.values(this.stages());
    const total = values.reduce((sum, val) => sum + val, 0);
    return Math.round(total / values.length);
  });

  getStageStatus(stage: string): string {
    const percent = this.stages()[stage];
    if (percent === 100) return 'success';
    if (percent > 0) return 'normal';
    return 'normal';
  }
}
```

    if (disk > 60) return '#faad14';
    return '#52c41a';
  }
}

## Best Practices | Thực Hành Tốt Nhất

1. **Show meaningful progress** | **Hiển thị tiến trình có ý nghĩa**: Update progress based on actual completion, not time

2. **Provide context** | **Cung cấp ngữ cảnh**: Show what's being progressed (file name, task name, etc.)

3. **Use appropriate type** | **Dùng loại phù hợp**: Line for simple tasks, circle for dashboards or compact spaces

4. **Color feedback** | **Phản hồi màu sắc**: Use status colors to indicate success/error states

5. **Allow cancellation** | **Cho phép hủy**: Provide a way to cancel long-running operations
