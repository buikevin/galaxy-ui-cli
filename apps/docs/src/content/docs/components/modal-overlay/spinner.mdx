The **Spinner** component provides visual feedback during loading states with multiple animation styles. Perfect for indicating background processes or data fetching.

**Spinner** là component cung cấp phản hồi trực quan trong khi tải với nhiều kiểu hoạt ảnh. Hoàn hảo để chỉ báo các quy trình nền hoặc tải dữ liệu.

## Features | Tính Năng

- **5 animation types** | **5 loại hoạt ảnh**: Spin, pulse, bounce, dots, bars
- **Size variants** | **Biến thể kích thước**: Tiny, small, medium, large, huge
- **Color customization** | **Tùy chỉnh màu**: Primary, success, warning, danger, custom colors
- **Optional text** | **Văn bản tùy chọn**: Loading text alongside spinner
- **Overlay mode** | **Chế độ lớp phủ**: Full-screen or container overlay
- **Dark mode** | **Chế độ tối**: Full dark mode support

## Props | Thuộc Tính

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `type` | `'spin' \| 'pulse' \| 'bounce' \| 'dots' \| 'bars'` | `'spin'` | Animation type <br/> Loại hoạt ảnh |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg' \| 'xl'` | `'md'` | Spinner size <br/> Kích thước spinner |
| `color` | `'primary' \| 'success' \| 'warning' \| 'danger'` | `'primary'` | Spinner color <br/> Màu spinner |
| `customColor` | `string` | - | Custom CSS color <br/> Màu CSS tùy chỉnh |
| `text` | `string` | - | Loading text <br/> Văn bản tải |
| `textPosition` | `'below' \| 'inline'` | `'below'` | Text position <br/> Vị trí văn bản |

### Button Loading States | Trạng Thái Tải Nút

```typescript
import { Component, signal } from '@angular/core';

@Component({
  template: `
    <div class="space-x-4">
      <!-- Primary button loading | Nút chính đang tải -->
      <button
        (click)="handleSave()"
        [disabled]="isSaving()"
        class="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700 disabled:opacity-50 flex items-center gap-2">
        @if (isSaving()) {
          <g-spinner size="sm" customColor="white" type="spin" />
          <span>Saving...</span>
        } @else {
          <span>Save</span>
        }
      </button>

      <!-- Danger button loading | Nút nguy hiểm đang tải -->
      <button
        (click)="handleDelete()"
        [disabled]="isDeleting()"
        class="px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700 disabled:opacity-50 flex items-center gap-2">
        @if (isDeleting()) {
          <g-spinner size="sm" customColor="white" type="dots" />
          <span>Deleting...</span>
        } @else {
          <span>Delete</span>
        }
      </button>

      <!-- Success button loading | Nút thành công đang tải -->
      <button
        (click)="handleSubmit()"
        [disabled]="isSubmitting()"
        class="px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700 disabled:opacity-50 flex items-center gap-2">
        @if (isSubmitting()) {
          <g-spinner size="sm" customColor="white" type="pulse" />
          <span>Submitting...</span>
        } @else {
          <span>Submit</span>
        }
      </button>
    </div>
  `
})
export class ButtonLoadingStates {
  isSaving = signal(false);
  isDeleting = signal(false);
  isSubmitting = signal(false);

  async handleSave() {
    this.isSaving.set(true);
    await this.api.save();
    this.isSaving.set(false);
  }

  async handleDelete() {
    this.isDeleting.set(true);
    await this.api.delete();
    this.isDeleting.set(false);
  }

  async handleSubmit() {
    this.isSubmitting.set(true);
    await this.api.submit();
    this.isSubmitting.set(false);
  }
}
```

### Infinite Scroll Loading | Tải Cuộn Vô Hạn

```typescript
@Component({
  template: `
    <div class="space-y-4">
      @for (item of items(); track item.id) {
        <div class="border rounded-lg p-4">
          <h3 class="font-semibold">{{ item.title }}</h3>
          <p class="text-gray-600">{{ item.description }}</p>
        </div>
      }

      <!-- Load more spinner | Spinner tải thêm -->
      @if (isLoadingMore()) {
        <div class="flex justify-center py-8">
          <g-spinner
            type="dots"
            text="Loading more..."
            color="primary" />
        </div>
      }

      <!-- End of list | Hết danh sách -->
      @if (!hasMore() && items().length > 0) {
        <div class="text-center py-8 text-gray-500">
          <p>No more items to load | Không còn mục để tải</p>
        </div>
      }
    </div>

    <!-- Scroll trigger -->
    <div #scrollTrigger></div>
  `
})
export class InfiniteScrollLoading implements OnInit, AfterViewInit {
  @ViewChild('scrollTrigger') scrollTrigger!: ElementRef;

  items = signal([]);
  isLoadingMore = signal(false);
  hasMore = signal(true);
  page = signal(1);

  ngAfterViewInit() {
    // Setup intersection observer
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && !this.isLoadingMore() && this.hasMore()) {
          this.loadMore();
        }
      },
      { threshold: 0.1 }
    );

    observer.observe(this.scrollTrigger.nativeElement);
  }

  async loadMore() {
    this.isLoadingMore.set(true);

    try {
      const newItems = await this.api.getItems(this.page());

      if (newItems.length === 0) {
        this.hasMore.set(false);
      } else {
        this.items.update(current => [...current, ...newItems]);
        this.page.update(p => p + 1);
      }
    } finally {
      this.isLoadingMore.set(false);
    }
  }
}
```

### Search Loading | Tải Tìm Kiếm

```typescript
import { Component, signal, effect } from '@angular/core';
import { debounceTime, distinctUntilChanged } from 'rxjs/operators';

@Component({
  template: `
    <div class="space-y-4">
      <!-- Search input | Ô tìm kiếm -->
      <div class="relative">
        <input
          type="text"
          [(ngModel)]="searchQuery"
          placeholder="Search users... | Tìm kiếm người dùng..."
          class="w-full border rounded-lg pl-10 pr-10 py-2" />

        <!-- Search icon -->
        <svg class="absolute left-3 top-2.5 w-5 h-5 text-gray-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z"/>
        </svg>

        <!-- Loading spinner in input -->
        @if (isSearching()) {
          <div class="absolute right-3 top-2.5">
            <g-spinner size="xs" type="spin" />
          </div>
        }
      </div>

      <!-- Results -->
      <div class="border rounded-lg">
        @if (isSearching()) {
          <div class="flex justify-center py-12">
            <g-spinner
              type="pulse"
              text="Searching..."
              color="primary" />
          </div>
        } @else if (searchResults().length === 0 && searchQuery()) {
          <div class="text-center py-12 text-gray-500">
            <p>No results found for "{{ searchQuery() }}"</p>
          </div>
        } @else if (searchResults().length > 0) {
          <div class="divide-y">
            @for (result of searchResults(); track result.id) {
              <div class="p-4 hover:bg-gray-50">
                <h3 class="font-semibold">{{ result.name }}</h3>
                <p class="text-sm text-gray-600">{{ result.email }}</p>
              </div>
            }
          </div>
        }
      </div>
    </div>
  `
})
export class SearchLoading {
  searchQuery = signal('');
  searchResults = signal([]);
  isSearching = signal(false);

  constructor() {
    // Watch search query changes
    effect(() => {
      const query = this.searchQuery();
      if (query.length >= 2) {
        this.performSearch(query);
      } else {
        this.searchResults.set([]);
      }
    });
  }

  async performSearch(query: string) {
    this.isSearching.set(true);

    // Debounce search
    await new Promise(resolve => setTimeout(resolve, 500));

    try {
      const results = await this.api.search(query);
      this.searchResults.set(results);
    } finally {
      this.isSearching.set(false);
    }
  }
}
```

## Best Practices | Thực Hành Tốt Nhất

1. **Appropriate size** | **Kích thước phù hợp**: Use small spinners for inline, larger for full-screen

2. **Add context** | **Thêm ngữ cảnh**: Include descriptive text when possible (e.g., "Loading data...")

3. **Don't overuse** | **Đừng lạm dụng**: Show spinners only for operations > 300ms

4. **Reduced motion** | **Giảm chuyển động**: Respect user's motion preferences

5. **Color contrast** | **Tương phản màu**: Ensure spinner is visible against background
