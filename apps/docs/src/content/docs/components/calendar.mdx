The **Calendar** component provides a beautiful and functional date picker with month/year navigation, date range restrictions, and multiple status colors.

## Features

- Interactive month and year navigation
- Month and year picker views
- Min/max date constraints
- Today button
- Multiple status colors
- Dark mode support
- Keyboard accessible

## Installation

```bash
npx galaxy-ui-cli@latest add calendar
```

## Usage

### Basic Usage

```typescript
import { Component, signal } from '@angular/core';
import { CalendarComponent } from '@/components/ui/calendar';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [CalendarComponent],
  template: `
    <galaxy-calendar
      [calendarSelectedDate]="selectedDate()"
      (selectedDateChange)="selectedDate.set($event)">
    </galaxy-calendar>

    <p>Selected: {{ selectedDate() | date }}</p>
  `
})
export class ExampleComponent {
  selectedDate = signal<Date | null>(null);
}
```

### With Date Range Restrictions

```typescript
@Component({
  template: `
    <galaxy-calendar
      [calendarSelectedDate]="selectedDate()"
      [calendarMinDate]="minDate"
      [calendarMaxDate]="maxDate"
      (selectedDateChange)="selectedDate.set($event)">
    </galaxy-calendar>
  `
})
export class DateRangeExample {
  selectedDate = signal<Date | null>(null);

  // Only allow dates in the next 30 days
  minDate = new Date();
  maxDate = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000);
}
```

### Different Statuses

```html
<galaxy-calendar [calendarStatus]="'primary'"></galaxy-calendar>
<galaxy-calendar [calendarStatus]="'success'"></galaxy-calendar>
<galaxy-calendar [calendarStatus]="'info'"></galaxy-calendar>
<galaxy-calendar [calendarStatus]="'warning'"></galaxy-calendar>
<galaxy-calendar [calendarStatus]="'danger'"></galaxy-calendar>
```

### Without Today Button

```html
<galaxy-calendar [calendarShowTodayButton]="false"></galaxy-calendar>
```

## Props

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `calendarSelectedDate` | `Date \| null` | `null` | Currently selected date |
| `calendarMinDate` | `Date \| null` | `null` | Minimum selectable date |
| `calendarMaxDate` | `Date \| null` | `null` | Maximum selectable date |
| `calendarStatus` | `CalendarStatus` | `'primary'` | Status color theme |
| `calendarShowTodayButton` | `boolean` | `true` | Show/hide today button |

## Events

| Event | Type | Description |
|-------|------|-------------|
| `selectedDateChange` | `EventEmitter<Date \| null>` | Emitted when date is selected |

### Appointment Booking

```typescript
@Component({
  template: `
    <div class="max-w-sm">
      <h3>Book an Appointment</h3>

      <galaxy-calendar
        [calendarSelectedDate]="appointmentDate()"
        [calendarMinDate]="tomorrow()"
        [calendarMaxDate]="maxBookingDate()"
        [calendarStatus]="'primary'"
        (selectedDateChange)="selectAppointment($event)">
      </galaxy-calendar>

      @if (appointmentDate()) {
        <div class="mt-4">
          <p class="font-medium">Available times on {{ appointmentDate() | date }}:</p>
          <div class="grid grid-cols-3 gap-2 mt-2">
            @for (time of availableTimes; track time) {
              <button
                class="px-3 py-2 border rounded-md hover:bg-violet-50"
                (click)="selectTime(time)">
                {{ time }}
              </button>
            }
          </div>
        </div>
      }
    </div>
  `
})
export class AppointmentBookingExample {
  appointmentDate = signal<Date | null>(null);
  selectedTime = signal<string | null>(null);

  tomorrow = computed(() => {
    const date = new Date();
    date.setDate(date.getDate() + 1);
    return date;
  });

  maxBookingDate = computed(() => {
    const date = new Date();
    date.setDate(date.getDate() + 60); // 60 days in advance
    return date;
  });

  availableTimes = ['09:00', '10:00', '11:00', '14:00', '15:00', '16:00'];

  selectAppointment(date: Date | null) {
    this.appointmentDate.set(date);
    this.selectedTime.set(null);
  }

  selectTime(time: string) {
    this.selectedTime.set(time);
    console.log('Booked:', this.appointmentDate(), time);
  }
}
```

### Birthday Selector

```typescript
@Component({
  template: `
    <galaxy-calendar
      [calendarSelectedDate]="birthday()"
      [calendarMaxDate]="today"
      [calendarStatus]="'info'"
      (selectedDateChange)="birthday.set($event)">
    </galaxy-calendar>

    @if (birthday()) {
      <p>Age: {{ calculateAge(birthday()!) }} years</p>
    }
  `
})
export class BirthdaySelectorExample {
  birthday = signal<Date | null>(null);
  today = new Date();

  calculateAge(birthDate: Date): number {
    const today = new Date();
    let age = today.getFullYear() - birthDate.getFullYear();
    const monthDiff = today.getMonth() - birthDate.getMonth();

    if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {
      age--;
    }

    return age;
  }
}
```

## Features

### Month Picker

Click on the month name to open a quick month selector grid.

### Year Picker

Click on the year to open a scrollable year selector.

### Navigation

- Use arrow buttons to navigate months
- Navigation is automatically disabled when reaching min/max dates

### Today Button

Quick access to jump to today's date and select it.

## Accessibility

- Keyboard navigation support
- Proper ARIA labels
- Focus management
- Screen reader friendly
- High contrast support

## Styling

The calendar inherits the theme from the status prop:
- **Primary**: Violet theme (default)
- **Success**: Emerald theme
- **Info**: Cyan theme
- **Warning**: Amber theme
- **Danger**: Red theme
- **Basic**: Slate theme

Each theme applies to:
- Selected date background
- Today's date border
- Month/year picker highlights
- Focus rings
