The **Select** component provides a feature-rich dropdown with custom styling, search support, and keyboard navigation.

## Features

- Multiple status colors
- Five size variants
- Searchable options
- Label, hint, and error support
- Keyboard navigation
- Dark mode support
- Custom option rendering

## Installation

```bash
npx galaxy-ui-cli@latest add select
```

## Usage

### Basic Usage

```typescript
import { Component, signal } from '@angular/core';
import { SelectComponent, type SelectOption } from '@/components/ui/select';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [SelectComponent],
  template: `
    <galaxy-select
      [selectLabel]="'Country'"
      [selectOptions]="countries"
      [selectSelected]="selectedCountry()"
      (selectedChange)="selectedCountry.set($event)">
    </galaxy-select>
  `
})
export class ExampleComponent {
  selectedCountry = signal(null);

  countries: SelectOption[] = [
    { label: 'United States', value: 'us' },
    { label: 'Canada', value: 'ca' },
    { label: 'United Kingdom', value: 'uk' },
    { label: 'Germany', value: 'de' },
    { label: 'France', value: 'fr' },
  ];
}
```

## Props

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `selectLabel` | `string` | `''` | Label text |
| `selectPlaceholder` | `string` | `'Select an option'` | Placeholder text |
| `selectHint` | `string` | `''` | Hint text below select |
| `selectError` | `string` | `''` | Error message |
| `selectOptions` | `SelectOption[]` | `[]` | Array of options |
| `selectSelected` | `any` | `null` | Selected value |
| `selectStatus` | `SelectStatus` | `'basic'` | Status color |
| `selectSize` | `SelectSize` | `'medium'` | Size variant |
| `selectShape` | `SelectShape` | `'rectangle'` | Border radius |
| `selectDisabled` | `boolean` | `false` | Disabled state |
| `selectRequired` | `boolean` | `false` | Required field |
| `selectSearchable` | `boolean` | `false` | Enable search |
| `selectFullWidth` | `boolean` | `true` | Full width |

## SelectOption Interface

```typescript
interface SelectOption {
  label: string;      // Display text
  value: any;         // Value to be stored
  disabled?: boolean; // Optional disabled state
}
```

## Events

| Event | Type | Description |
|-------|------|-------------|
| `selectedChange` | `EventEmitter<any>` | Emitted when selection changes |

## Advanced Example

```typescript
@Component({
  template: `
    <form [formGroup]="form">
      <galaxy-select
        [selectLabel]="'Department'"
        [selectOptions]="departments"
        [selectSearchable]="true"
        [selectRequired]="true"
        [selectError]="departmentError()"
        formControlName="department">
      </galaxy-select>

      <galaxy-select
        [selectLabel]="'Team'"
        [selectOptions]="filteredTeams()"
        [selectDisabled]="!form.get('department')?.value"
        formControlName="team">
      </galaxy-select>
    </form>
  `
})
export class DependentSelectsExample {
  form = new FormGroup({
    department: new FormControl(null, Validators.required),
    team: new FormControl(null),
  });

  departments: SelectOption[] = [
    { label: 'Engineering', value: 'eng' },
    { label: 'Marketing', value: 'mkt' },
    { label: 'Sales', value: 'sales' },
  ];

  teams = {
    eng: [
      { label: 'Frontend', value: 'frontend' },
      { label: 'Backend', value: 'backend' },
    ],
    mkt: [
      { label: 'Digital', value: 'digital' },
      { label: 'Content', value: 'content' },
    ],
    sales: [
      { label: 'Enterprise', value: 'enterprise' },
      { label: 'SMB', value: 'smb' },
    ],
  };

  filteredTeams = computed(() => {
    const dept = this.form.get('department')?.value;
    return dept ? this.teams[dept as keyof typeof this.teams] : [];
  });

  departmentError() {
    const control = this.form.get('department');
    return control?.hasError('required') ? 'Please select a department' : '';
  }
}
```

## Keyboard Navigation

- **Enter/Space**: Open dropdown
- **Arrow Up/Down**: Navigate options
- **Enter**: Select highlighted option
- **Escape**: Close dropdown
- **Type**: Search (when searchable)
